BCAPI.Models.WebApp. CustomField

new CustomField()

This class provides support for custom fields description belonging to BCAPI.Models.WebApp.App

Create a new custom field

var customField = new BCAPI.Models.WebApp.CustomField("Test webapp", {
    "name": "Part code",
    "type": "DataSource",
    "listItems": null,
    "dataSourceName": "Part Codes",
    "required": false,
    "order": 1
});

customField.save({
    success: function(fieldModel) {
        // do something on success.
    }
});
